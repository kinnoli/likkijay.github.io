String methods

var x = "some string";
var y = "Some other string";

-- charAt(index): shows what character is present at given index.

x.charAt(1); // outputs "o"

-- charCodeAt(index): shows unicode of a character at given index.

x.charCodeAt(1); // outputs "111"

-- concat(): concatenate 2 strings.

x.concat(y); // outputs "some stringSome other string"

-- endsWith(): checks if string ends with given characters or substring.

x.endsWith("ing"); // outputs "true"

-- String.fromCharCode(): shows what character is present at given unicode position.

String.fromCharCode(111); // outputs "o"

-- includes(): shows if string contains given substring or character.

x.includes("som"); // outputs "true"

-- indexOf: shows index of given character (the first that appears in the string).

x.indexOf("o"); outputs "1"

-- lastIndexOf(): shows the last index of given character

x.lastIndexOf("s"); // outputs "5"

-- match(): returns an array of matched characters if they present in the string. Accepts arguments in form of regular expressions: /regexp/g.

x.match(/ome/g); outputs "["ome"]"

-- repeat(): returns the string given number of times.

x.repeat(2); outputs "some stringsome string"

-- replace(): returns a new string with replaced characters or substrings or regular expressions.

x.replace("some", "SOME"); outputs "SOME string"
x.replace(/s/g, "S"); outputs "Some String"

-- search(): returns index/indexes of provided arguments if present.

x.search("s"); // outputs "0"

-- slice(): returns a new string starting at given index and ending at index before provided end point.

x.slice(0, 5); // outputs "some "

-- split(): splits a string into an array, divider is defined as argument.

x.split(" "); outputs "["some", "string"]"

-- startsWith(): checks if string starts with given character or substring.

x.startsWith("some"); // outputs "true"

-- substr(): returns a string of characters begins with given index and through to given number of characters.

x.substr(2, 4); // outputs "me s"

-- substring(): returns a string starting with given index and ending before given second index (like slice()).

x.substring(2, 4); // outputs "me"

-- toLowerCase(): returns a string with lower case characters.

y.toLowerCase(); // outputs "some other string"

-- toUpperCase(): returns a string with upper case characters.

x.toUpperCase(); // outputs "SOME STRING"

-- trim(): removes white spaces from left and rigth sides of given string.

var c = " some string ";
c.trim(); // outputs "some string"
